 * @param {number[]} nums2
 * @return {number[]}

var intersect = function (nums1, nums2) {
    nums1.sort((a, b) => a - b)
    nums2.sort((a, b) => a - b)
    let left = 0, right = 0, res = []
    while (left < nums1.length && right < nums2.length) {
        if (nums1[left] === nums2[right]) {
            res.push(nums1[left])
            left++
            right++
        } else {
            nums1[left] > nums2[right] ? right++ : left++
        }
    }
    return res
};
